CREATE DATABASE calensenai;
USE calensenai;

CREATE TABLE alunos(
id INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
nome VARCHAR(255) NOT NULL,
email VARCHAR(255) NOT NULL,
senha VARCHAR(255) NOT NULL,
matricula INT NOT NULL
);

CREATE TABLE professores(
id INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
nome VARCHAR(255) NOT NULL,
email VARCHAR(255) NOT NULL,
senha VARCHAR(255) NOT NULL,
idp INT NOT NULL
);

CREATE TABLE cordenacao(
id INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
nome VARCHAR(255) NOT NULL,
email VARCHAR(255) NOT NULL,
senha VARCHAR(255) NOT NULL
);

CREATE TABLE horarios(
id INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
horario_data DATE,
horario_hora time,
horario_professor VARCHAR(255),
horario_curso VARCHAR(255)
);

CREATE TABLE turmas(
id INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
curso VARCHAR(255),
aluno VARCHAR(255),
periodo VARCHAR(255)
);

CREATE TABLE localizacao(
id INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
instituicao VARCHAR(255),
sala VARCHAR(3),
professor VARCHAR(255),
turma VARCHAR(255),
itens VARCHAR(255),
tipo_sala VARCHAR(255)
);
